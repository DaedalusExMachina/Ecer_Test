Cleaning and Preprocessing

Cleaning function

clean_func <- function(df_c){

        df <- df_c
    
        df$text <-  gsub("©.*", " ", df$text)
        df$text <-  sub("\\¸.*"," ", df$text, perl = TRUE)
        df$text <-  gsub('[0-9]+', '', df$text)
        df$text <-  gsub('<0-9>+', '', df$text)
        #df$text    <-  gsub("['s]", '', df$text)
        df$text <-  gsub("[']", '', df$text)
        df$text <-  gsub("'", '', df$text)
        df$text     <-  sub("\\s*\\([^\\)]+\\)","",df$text)
        df$text     <-  gsub("\\[[^][]*]", "", df$text)
        df$text <-  stringi::stri_trans_general(df$text, "latin-ascii")

        text_un <- df %>% unnest_tokens(word, text)
        text_sto <- text_un  %>% anti_join(stop)
        text_stem <- text_sto %>% mutate(word=wordStem(word))
        text_stemsto <-     text_stem %>% anti_join(stop)

        text_renest <- text_stemsto %>% group_by(ident) %>% dplyr::summarise(text = paste(word, collapse = " ")) %>% ungroup() 
        df_c <- text_renest
        
    }
